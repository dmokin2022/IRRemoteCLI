void keytable_output_item(uint8_t i) {
    Serial.print("id = "); Serial.print(key_table[i].id); Serial.print(" ");
    Serial.print("code = "); Serial.print(key_table[i].code); Serial.print(" ");
    //Serial.print("name = "); Serial.print(key_table[i].name); Serial.print(" ");
    Serial.print("mode = "); Serial.print(key_table[i].mode); Serial.print(" ");
    Serial.print("pin = "); Serial.print(key_table[i].pin); Serial.print(" ");
    Serial.print("val = "); Serial.print(key_table[i].var_val); Serial.print(" ");
    Serial.print("step = "); Serial.print(key_table[i].var_step); Serial.print(" ");
    Serial.print("var_id = "); Serial.print(key_table[i].var_id); Serial.print(" ");
    Serial.println();
}

// Вывод всей таблицы

    }
